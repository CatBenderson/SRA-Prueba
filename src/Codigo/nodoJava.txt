package arbolJava;

public class Nodo {
    public Nodo padre;
    public Nodo derecha; 
    public Nodo izquierda;
    public int servicio;
    public String nombre;

    public Nodo(int indice){
        this.servicio = indice;
        this.derecha=null;
        this.izquierda = null;
        this.padre = null;
        this.nombre = null;
    }
}
